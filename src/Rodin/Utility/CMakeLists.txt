set(RodinUtility_HEADERS
  IsSpecialization.h
  OptionalReference.h
  Overloaded.h
  Misc.h
  )

set(RodinUtility_SRCS
  Misc.cpp
  )

add_library(RodinUtility
  ${RodinUtility_SRCS} ${RodinUtility_HEADERS})

  INTERFACE
  $<TARGET_PROPERTY:Rodin,INTERFACE_INCLUDE_DIRECTORIES>
  PUBLIC
  ${MFEM_INCLUDE_DIRS}
  )

target_link_libraries(RodinUtility
  PRIVATE
  mfem)
